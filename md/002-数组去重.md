```javascript
// Set 数组去重

Array.prototype.myUnique = function () {
  return [...new Set(this)];
};

const arr = [
  '',
  '',
  0,
  -0, // 参考(mdn/.../Set#值的相等)一节
  123,
  123,
  NaN, // 参考(mdn/.../Set#值的相等)一节
  NaN,
  undefined,
  undefined,
  null,
  null,
  {},
  {},
  [],
  [],
  true,
  true,
  () => {},
  () => {},
  Symbol(),
  Symbol('cc'),
  Symbol('cc'),
];

console.log(arr.myUnique().length); // 16

```
