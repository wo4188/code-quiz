// 指定范围的所有质数
function getPrimeNums(min, max) {
  function _isPrime(num) {
    if (num <= 1) return false;

    // '2' 是唯一的偶质数(既是偶数又是质数的自然数)
    if (num === 2) return true;

    if (num % 2 === 0) return false;

    const sqrt = Math.sqrt(num);
    for (let i = 3; i < sqrt; i += 2) {
      if (num % i === 0) return false;
    }

    return true;
  }

  const primeNums = [];
  for (let i = Math.max(min, 2); i <= max; i++) {
    if (_isPrime(i)) {
      primeNums.push(i);
    }
  }

  return primeNums;
}

getPrimeNums(1, 10); // [2, 3, 5, 7, 9]
